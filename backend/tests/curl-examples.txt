## TransitOps Auth API Test Examples

# 1. Signup - Create a fleet manager
curl -X POST http://localhost:5000/api/auth/signup \
  -H "Content-Type: application/json" \
  -d '{
    "name": "John Doe",
    "email": "john@example.com",
    "password": "password123",
    "role": "fleet_manager"
  }'

# 2. Signup - Create a driver (will fail with EMAIL_EXISTS if using same email)
curl -X POST http://localhost:5000/api/auth/signup \
  -H "Content-Type: application/json" \
  -d '{
    "name": "Jane Smith",
    "email": "jane@example.com",
    "password": "password123",
    "role": "driver"
  }'

# 3. Login - Get JWT token (replace with actual email/password)
curl -X POST http://localhost:5000/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{
    "email": "john@example.com",
    "password": "password123"
  }'

# 4. Get Current User (me) - Replace <YOUR_TOKEN> with token from login
curl -X GET http://localhost:5000/api/auth/me \
  -H "Authorization: Bearer <YOUR_TOKEN>"

# 5. Logout - Replace <YOUR_TOKEN>
curl -X POST http://localhost:5000/api/auth/logout \
  -H "Authorization: Bearer <YOUR_TOKEN>"
